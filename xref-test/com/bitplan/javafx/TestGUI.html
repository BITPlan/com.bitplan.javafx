71    @Test
72    public void testNumberMatchers() {
73      String texts[]= {"1","a","1.2","2,5","1.,2"};
74      boolean iexpected[]= {true,false,false,false,false};
75      boolean dexpected[]= {true,false,true,true,false};
76      Pattern pi=Pattern.compile(GenericControl.INTEGER_MATCH);
77      Pattern pd=Pattern.compile(GenericControl.DOUBLE_MATCH);
78        
79      int i=0;
80      for (String text:texts) {
81        Matcher matcheri=pi.matcher(text);
82        Matcher matcherd=pd.matcher(text);
83        assertEquals("integer "+text,iexpected[i],matcheri.matches());
84        assertEquals("double "+text,dexpected[i],matcherd.matches());
85        i++;
86      }
87    }

112   @Test
113   public void testIcons() throws Exception {
114     GlyphFont fontAwesome = GlyphFontRegistry.font("FontAwesome");
115     ConstrainedGridPane gridPane = new ConstrainedGridPane();
116     org.controlsfx.glyphfont.FontAwesome.Glyph[] sglyphs = FontAwesome.Glyph
117         .values();
118     List<Node> icons = new ArrayList<Node>();
119     for (FontAwesome.Glyph glyph : sglyphs) {
120       Glyph icon = fontAwesome.create(glyph);
121       icons.add(icon);
122       icon.setId(glyph.name());
123       icon.setFontSize(24);
124       icon.setTooltip(new Tooltip(glyph.name()));
125     }
126     XYTabPane xyTabPane = new XYTabPane(24);
127     for (Entry<String, Node> iconentry : xyTabPane.getIconMap().entrySet()) {
128       icons.add(iconentry.getValue());
129       iconentry.getValue().setId(iconentry.getKey());
130     }
131     int i = 0;
132     for (Node icon : icons) {
133       int row = i / 30;
134       int col = i % 30;
135       i++;
136       Button button = new Button();
137       button.setGraphic(icon);
138       button.setTooltip(new Tooltip(icon.getId()));
139       gridPane.add(button, col, row);
140     }
141     SampleApp.createAndShow("icons", gridPane, SHOW_TIME);
142   }

144   @Test
145   public void testVerticalTabs() throws Exception {
146     // https://stackoverflow.com/questions/16708578/javafx-2-0-tabpane-tabs-at-left-and-keep-tab-header-horizontal
147     XYTabPane xyTabPane = new XYTabPane(64);
148 
149     String[] rowNames = { "DASHBOARD", "PLUG", "exchange", "EYE", "GEAR" };
150     String[] colNames = { "ROAD", "LOCK", "CAMERA", "MAP_MARKER", "QUESTION",
151         "SIGNAL", "AREA_CHART", "BAR_CHART", "LINE_CHART", "KEY",
152         "FILE_TEXT_ALT", "CLOCK_ALT", "POWER_OFF", "INFO", "ADJUST", "BOLT",
153         "PLAY", "CHECK", "DASHBOARD", "EDIT", "FILE_PHOTO_ALT", "FILE_TEXT_ALT",
154         "FLAG_CHECKERED", "MALE", "SMILE_ALT", "MOON_ALT", "SEARCH_MINUS",
155         "SEARCH_PLUS", "SLIDERS", "CHAIN", "EDIT", "CAR", "USER", "WIFI",
156         "WRENCH", "battery0" };
157     int col = 0;
158     for (String rowName : rowNames) {
159       TabPane hTabPane = xyTabPane.addTabPane(rowName, rowName, rowName);
160       for (int i = 0; i < 7; i++) {
161         xyTabPane.addTab(hTabPane, colNames[col], null, colNames[col++],
162             new Pane());
163       }
164     }
165     // xyTabPane.tabMap.get("battery25").getGraphic().setRotate(90);
166     SampleApp sampleApp = new SampleApp("vertical tabs", xyTabPane);
167     sampleApp.show();
168     sampleApp.waitOpen();
169     Set<String> tabids = xyTabPane.getTabMap().keySet();
170     for (String tabId : tabids) {
171       xyTabPane.selectTab(tabId);
172       Thread.sleep(SHOW_TIME / tabids.size());
173     }
174     sampleApp.close();
175   }
