41  public class TranslucentPane extends Application {
42    @Override
43    public void start(final Stage stage) throws Exception {
44      final ImageView imageView = new ImageView(new Image(
45          "https://upload.wikimedia.org/wikipedia/commons/b/b7/Idylls_of_the_King_3.jpg"));
46      imageView.setFitHeight(300);
47      imageView.setFitWidth(228);
48  
49      final Label label = new Label("The Once\nand\nFuture King");
50      label.setStyle(
51          "-fx-text-fill: goldenrod; -fx-font: italic 20 \"serif\"; -fx-padding: 0 0 20 0; -fx-text-alignment: center");
52  
53      StackPane glass = new StackPane();
54      StackPane.setAlignment(label, Pos.BOTTOM_CENTER);
55      glass.getChildren().addAll(label);
56      glass.setStyle(
57          "-fx-background-color: rgba(0, 100, 100, 0.5); -fx-background-radius: 10;");
58      glass.setMaxWidth(imageView.getFitWidth() - 40);
59      glass.setMaxHeight(imageView.getFitHeight() - 40);
60      /*Line line = new Line(0,0,100,10);
61      line.setStrokeWidth(2);
62      line.setStroke(Color.BLACK);
63      glass.getChildren().add(line);*/
64  
65      final StackPane layout = new StackPane();
66      layout.getChildren().addAll(imageView, glass);
67      layout.setStyle("-fx-background-color: silver; -fx-padding: 10;");
68      stage.setScene(new Scene(layout));
69      stage.show();
70    }
71    public static void main(String[] args) {
72      launch(args);
73    }
74  }
