24  package com.bitplan.javafx;

26  import java.net.URL;
27  import java.util.Optional;
28  
29  import javafx.application.Platform;
30  import javafx.geometry.Insets;
31  import javafx.scene.Node;
32  import javafx.scene.control.ButtonBar.ButtonData;
33  import javafx.scene.control.ButtonType;
34  import javafx.scene.control.Dialog;
35  import javafx.scene.control.Label;
36  import javafx.scene.control.PasswordField;
37  import javafx.scene.control.TextField;
38  import javafx.scene.image.ImageView;
39  import javafx.scene.layout.GridPane;
40  import javafx.util.Pair;
41  
42  /**
43   * Login Dialog
44   * @author wf
45   * see http://code.makery.ch/blog/javafx-dialogs-official/
46   */
47  public class LoginDialog {
48    /**
49     * show the Login dialog
50     * @return the pair
51     */
52    public Optional<Pair<String, String>> show() {
53      // Create the custom dialog.
54      Dialog<Pair<String, String>> dialog = new Dialog<>();
55      dialog.setTitle("Login");
56      dialog.setHeaderText("Choose your credentials");
57  
58      // Set the icon (must be included in the project).
59      URL iconUrl = this.getClass().getResource("/icons/login.png");
60      if (iconUrl!=null)
61        dialog.setGraphic(new ImageView(iconUrl.toString()));
62  
63      // Set the button types.
64      ButtonType loginButtonType = new ButtonType("Login", ButtonData.OK_DONE);
65      dialog.getDialogPane().getButtonTypes().addAll(loginButtonType, ButtonType.CANCEL);
66  
67      // Create the username and password labels and fields.
68      GridPane grid = new GridPane();
69      grid.setHgap(10);
70      grid.setVgap(10);
71      grid.setPadding(new Insets(20, 150, 10, 10));
72  
73      TextField username = new TextField();
74      username.setPromptText("Username");
75      PasswordField password = new PasswordField();
76      password.setPromptText("Password");
77  
78      grid.add(new Label("Username:"), 0, 0);
79      grid.add(username, 1, 0);
80      grid.add(new Label("Password:"), 0, 1);
81      grid.add(password, 1, 1);
82  
83      // Enable/Disable login button depending on whether a username was entered.
84      Node loginButton = dialog.getDialogPane().lookupButton(loginButtonType);
85      loginButton.setDisable(true);
86  
87      // Do some validation (using the Java 8 lambda syntax).
88      username.textProperty().addListener((observable, oldValue, newValue) -> {
89          loginButton.setDisable(newValue.trim().isEmpty() || password.getText().isEmpty());
90      });
91      password.textProperty().addListener((observable, oldValue, newValue) -> {
92        loginButton.setDisable(newValue.trim().isEmpty() || username.getText().isEmpty());
93      });
94  
95      dialog.getDialogPane().setContent(grid);
96  
97      // Request focus on the username field by default.
98      Platform.runLater(() -> username.requestFocus());
99  
100     // Convert the result to a username-password-pair when the login button is clicked.
101     dialog.setResultConverter(dialogButton -> {
102         if (dialogButton == loginButtonType) {
103             return new Pair<>(username.getText(), password.getText());
104         }
105         return null;
106     });
107 
108     Optional<Pair<String, String>> result = dialog.showAndWait();
109     return result;
110   }
