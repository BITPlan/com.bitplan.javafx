121   /**
122    * get an initial Map of Icons
123    * 
124    * @return the icon map
125    */
126   public Map<String, Node> getIconMap() {
127     if (iconMap == null) {
128       iconMap = new HashMap<String, Node>();
129       char[] codes = { '\uf240', '\uf241', '\uf242', '\uf243', '\uf244',
130           '\uf0ec', '\uf2c9' };
131       String[] names = { "battery-full", "battery75", "battery50", "battery25",
132           "battery0", "exchange", "temp50" };
133       int i = 0;
134       for (char code : codes) {
135         Glyph icon = fontAwesome.create(code);
136         iconMap.put(names[i], icon);
137         icon.setTooltip(new Tooltip(names[i++]));
138       }
139     }
140     return iconMap;
141   }

143   /**
144    * create an XYTab Pane with the given iconSize
145    * 
146    * @param iconSize
147    *          - e.g. 48
148    */
149   public XYTabPane(int iconSize) {
150     super();
151     this.iconSize = iconSize;
152     setvTabPane(this.addTabPane("vTabPane"));
153     getvTabPane().setSide(Side.LEFT);
154     Tab filler = new Tab();
155     topLeftButton = new Button();
156     int tabSize = getTabSize();
157     topLeftButton.setMinSize(tabSize, tabSize);
158     topLeftButton.setMaxSize(tabSize, tabSize);
159     topLeftButton.setDisable(true);
160     filler.setGraphic(topLeftButton);
161     filler.setDisable(true);
162     getvTabPane().getTabs().add(filler);
163     this.addToMaps(filler, vTabPane);
164     fontAwesome = GlyphFontRegistry.font("FontAwesome");
165     super.getChildren().add(getvTabPane());
166   }

168   /**
169    * make clicked tabs blue
170    * 
171    * @param tabPane
172    */
173   protected void makeBlueOnSelect(TabPane tabPane) {
174     tabPane.getSelectionModel().selectedItemProperty()
175         .addListener((ov, oldTab, newTab) -> {
176           if (oldTab != null) {
177             setColor(oldTab, Color.BLUE, Color.BLACK);
178           }
179           setColor(newTab, Color.BLACK,Color.BLUE);
180         });
181   }

183   /**
184    * set the color of the graphic of the given tab
185    *
186    * @param tab - the tab to change
187    * @param oldColor - the old color
188    * @param newColor - the new color
189    */
190   protected void setColor(Tab tab, Color oldColor, Color newColor) {
191     Node graphic = tab.getGraphic();
192     
193     if (graphic instanceof Glyph) {
194       Glyph glyph = (Glyph) graphic;
195       glyph.setColor(newColor);
196     } else if (graphic instanceof ImageView) {
197       //  https://docs.oracle.com/javafx/2/image_ops/jfxpub-image_ops.htm
198       ImageView imageView=(ImageView) graphic;
199       imageView.setImage(reColor(imageView.getImage(),oldColor,newColor));
200     }
201   }

203   /**
204    * reColor the given InputImage from oldColor to the given newColor
205    * inspired by https://stackoverflow.com/a/12945629/1497139
206    * 
207    * @param inputImage
208    * @param oldColor
209    * @param newColor
210    * @return the reColored output Image
211    */
212   public static Image reColor(Image inputImage, Color oldColor, Color newColor) {
213     int W = (int) inputImage.getWidth();
214     int H = (int) inputImage.getHeight();
215     WritableImage outputImage = new WritableImage(W, H);
216     PixelReader reader = inputImage.getPixelReader();
217     PixelWriter writer = outputImage.getPixelWriter();
218     int ob=(int) oldColor.getBlue()*255;
219     int or=(int) oldColor.getRed()*255;
220     int og=(int) oldColor.getGreen()*255;
221     int nb=(int) newColor.getBlue()*255;
222     int nr=(int) newColor.getRed()*255;
223     int ng=(int) newColor.getGreen()*255;
224     for (int y = 0; y < H; y++) {
225       for (int x = 0; x < W; x++) {
226         int argb = reader.getArgb(x, y);
227         int a = (argb >> 24) & 0xFF;
228         int r = (argb >> 16) & 0xFF;
229         int g = (argb >>  8) & 0xFF;
230         int b =  argb        & 0xFF;
231         if (g==og && r==or && b==ob) {
232           r=nr;
233           g=ng;
234           b=nb;
235         }
236     
237         argb = (a << 24) | (r << 16) | (g << 8) | b;
238         writer.setArgb(x, y, argb);
239       }
240     }
241     return outputImage;
242   }

244   /**
245    * get the icon for the given name and fontSize
246    * 
247    * @param name
248    *          - the name of the glyph
249    * @param fontSize
250    *          - the fontSize of the glyph
251    * @return the Glyph
252    */
253   public Node getIcon(String name, int fontSize) {
254     Node icon = this.getIconMap().get(name + "x" + fontSize);
255     // if (icon == null) {
256     icon = this.getIconMap().get(name);
257     if (icon == null) {
258       try {
259         org.controlsfx.glyphfont.FontAwesome.Glyph fglyph = FontAwesome.Glyph
260             .valueOf(name.toUpperCase());
261         icon = fontAwesome.create(fglyph);
262       } catch (Throwable th) {
263         if (debug)
264           LOGGER.log(Level.INFO,
265               "Could not get FontAwesomeGlyph for icon " + name);
266       }
267     }
268     if (icon == null) {
269       URL iconUrl = this.getClass().getResource("/icons/" + name + ".png");
270       if (iconUrl != null) {
271         ImageView iconImage = new ImageView(iconUrl.toString());
272         iconImage.setFitHeight(fontSize);
273         iconImage.setFitWidth(fontSize);
274         iconImage.setId(name);
275         icon = iconImage;
276       } else {
277         if (debug)
278           LOGGER.log(Level.WARNING, "could not get resource for icon " + name);
279       }
280     }
281     if (icon != null) {
282       if (icon instanceof Glyph)
283         ((Glyph) icon).setFontSize(fontSize);
284       iconMap.put(name + "x" + fontSize, icon);
285     }
286     return icon;
287   }

311   /**
312    * add a tabPane with the given tabPane Id and iconName
313    * 
314    * @param title
315    * @param tabPaneId
316    *          / tabPaneId
317    * @param iconName
318    *          - the name of the icon
319    * @return - the tabPane
320    */
321   public TabPane addTabPane(String tabPaneId, String title, String iconName) {
322     TabPane tabPane = addTabPane(tabPaneId);
323     Tab tab = addTab(vTabPane, tabPaneId, title, iconName, tabPane);
324     vTabMap.put(tabPaneId, tab);
325     addToMaps(tab, tabPane);
326     return tabPane;
327   }

329   /**
330    * add the given tab/tabPane combination to the maps
331    * 
332    * @param tab
333    * @param tabPane
334    */
335   private void addToMaps(Tab tab, TabPane tabPane) {
336     this.vTabPaneMapByTab.put(tab, tabPane);
337     this.vTapMapByTabPane.put(tabPane, tab);
338   }

340   /**
341    * add a tab Pane with the given tabPaneId
342    * 
343    * @param tabPaneId
344    * @return the tabPane
345    */
346   public TabPane addTabPane(String tabPaneId) {
347     TabPane tabPane = new TabPane();
348     int tabSize = getTabSize();
349     tabPane.setTabMinHeight(tabSize);
350     tabPane.setTabMaxHeight(tabSize);
351     tabPane.setTabMinWidth(tabSize);
352     tabPane.setTabMaxWidth(tabSize);
353     // tabPane.setMaxWidth(Control.USE_PREF_SIZE);
354     // tabPane.setMaxHeight(Control.USE_PREF_SIZE);
355     // make sure it grows e.g. when Icons are set
356     // https://stackoverflow.com/a/25164425/1497139
357     VBox.setVgrow(tabPane, Priority.ALWAYS);
358     tabPane.setTabClosingPolicy(TabClosingPolicy.UNAVAILABLE);
359     // add click behavior
360     makeBlueOnSelect(tabPane);
361     // add tab Pane to map
362     this.tabPaneMap.put(tabPaneId, tabPane);
363     return tabPane;
364   }

390   /**
391    * add a tab
392    * 
393    * @param tabPane
394    *          the TabPane to add a Tab to
395    * @param index
396    * @param title
397    * @param glyphName
398    * @param content
399    * @return the tab
400    */
401   public Tab addTab(TabPane tabPane, String tabId, int index, String title,
402       String glyphName, Node content) {
403     Tab tab = new Tab();
404     if (glyphName != null) {
405       this.setTabGlyph(tab, glyphName);
406       getTabMap().put(tabId, tab);
407     }
408     if (title != null) {
409       tab.setTooltip(new Tooltip(title));
410       if (tab.getGraphic() == null) {
411         tab.setText(title);
412       }
413     }
414     tab.setContent(content);
415     if (index > 0)
416       tabPane.getTabs().add(index, tab);
417     else
418       tabPane.getTabs().add(tab);
419     this.tabPaneByTabIdMap.put(tabId, tabPane);
420     return tab;
421   }
