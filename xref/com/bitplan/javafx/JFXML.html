24  package com.bitplan.javafx;
25  
26  import java.net.URL;
27  import java.util.ResourceBundle;
28  
29  import com.bitplan.error.ErrorHandler;
30  import com.bitplan.error.ExceptionHandler;
31  import com.bitplan.gui.App;
32  import com.bitplan.i18n.Translator;
33  
34  import javafx.fxml.FXMLLoader;
35  import javafx.scene.Parent;
36  import javafx.stage.Stage;
37  
38  /**
39   * FXML loader utility that can integrate between the platform independent
40   * Application (App) concept and FXML
41   * 
42   * @author wf
43   *
44   */
45  public class JFXML {
46  
47    private String resourcePath;
48    private Stage stage;
49    private App app;
50  
51    public Stage getStage() {
52      return stage;
53    }
54  
55    public void setStage(Stage stage) {
56      this.stage = stage;
57    }
58  
59    public App getApp() {
60      return app;
61    }
62  
63    public void setApp(App app) {
64      this.app = app;
65    }
66  
67    /**
68     * create an FXML loader for the given resourcePath
69     * 
70     * @param resourcePath
71     */
72    public JFXML(String resourcePath, Stage stage, App app) {
73      this.stage = stage;
74      this.app = app;
75      this.resourcePath = resourcePath;
76    }
77  
78    /**
79     * Load a Result
80     * 
81     * @param <T>
82     */
83    public class LoadResult<T> {
84      Parent parent;
85      private T controller;
86  
87      public T getController() {
88        return controller;
89      }
90  
91      public void setController(T controller) {
92        this.controller = controller;
93      }
94    }
95  
96    /**
97     * load me from the given fxml fileName
98     * 
99     * @param fxmlFileName
100    * @return the load Result
101    */
102   public <T> LoadResult<T> load(String fxmlFileName) {
103     try {
104       LoadResult<T> loadResult = new LoadResult<T>();
105       ResourceBundle resourceBundle = Translator.getBundle();
106       URL fxml = JFXWizardPane.class
107           .getResource(resourcePath + fxmlFileName + ".fxml");
108       FXMLLoader fxmlLoader = new FXMLLoader(fxml, resourceBundle);
109       loadResult.parent = fxmlLoader.load();
110       loadResult.setController(fxmlLoader.getController());
111       return loadResult;
112     } catch (Throwable th) {
113       ErrorHandler.handle(th);
114     }
115     return null;
116   }
117 
118   /**
119    * load the presenter for the given entityName
120    * 
121    * @param entityName
122    * @param clazz
123    *          - the runtime class
124    * @param exceptionHandler
125    * @return the presenter
126    */
127   public <T extends BasePresenter<V>, V> T loadPresenter(String entityName,
128       Class<V> clazz, ExceptionHandler exceptionHandler) {
129     LoadResult<T> loadResult = load(entityName);
130     T presenter = loadResult.controller;
131     presenter.setParent(loadResult.parent);
132     presenter.init(stage, app, exceptionHandler);
133     presenter.setClazz(clazz);
134     return presenter;
135   }
136 }
