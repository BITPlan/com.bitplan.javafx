84    /**
85     * get the column Count estimate for the given field
86     * 
87     * @param field
88     * @return - the column count
89     */
90    public int getColumnCount(Field field) {
91      int columnCount = 0;
92      if (field.getFieldSize() != null) {
93        // heuristic correction of columns specified in generic description
94        // versus Java FX interpretation - the columns are too wide as of
95        // 2017-06-28
96        columnCount = field.getFieldSize() * 4 / 6;
97      }
98      return columnCount;
99    }

111   /**
112    * create a generic control for the given field
113    * 
114    * @param stage
115    * @param field
116    */
117   public GenericControl(Stage stage, Field field) {
118     this.stage = stage;
119     this.field = field;
120     String i18n=field.getLabel();
121     String ti18n=I18n.get(field.getI18nId());
122     if (ti18n!=null  && !"".equals(ti18n) && !field.getId().equals(ti18n))
123        i18n=ti18n;
124     String fieldType = field.getType();
125     if (fieldType == null || "Integer".equals(fieldType) || "Long".equals(fieldType)
126         || "Double".equals(fieldType) || "Password".equals(fieldType)
127         || "File".equals(fieldType) || "Directory".equals(fieldType)) {
128       if ("Password".equals(fieldType)) {
129         textField = new PasswordField();
130       } else {
131         textField = new TextField();
132       }
133       textField.setPromptText(i18n);
134       int columnCount = this.getColumnCount(field);
135       if (columnCount > 0)
136         textField.setPrefColumnCount(columnCount);
137       control = textField;
138     } else if ("Choice".equals(fieldType)) {
139       choiceBox = new ChoiceBox<String>();
140       choiceBox.getItems().addAll(field.getChoices().split(","));
141       control = choiceBox;
142     } else if ("Boolean".equals(fieldType)) {
143       setCheckBox(new CheckBox());
144       control = getCheckBox();
145     } else {
146       throw new IllegalArgumentException(
147           String.format("Unsupported field type %s", fieldType));
148     }
149     // force numeric content for integer fields
150     if ("Integer".equals(fieldType) || "Long".equals(fieldType) || "Double".equals(fieldType)) {
151       // force numeric content
152       // https://stackoverflow.com/a/36436243/1497139
153       String matchExpr = "";
154       switch (fieldType) {
155       case "Integer":
156       case "Long":
157         matchExpr = INTEGER_MATCH;
158         break;
159       case "Double":
160         matchExpr = DOUBLE_MATCH;
161         break;
162       }
163       final String match = matchExpr;
164       UnaryOperator<Change> filter = change -> {
165         String text = change.getText();
166 
167         if (text.matches(match)) {
168           return change;
169         }
170 
171         return null;
172       };
173       TextFormatter<String> textFormatter = new TextFormatter<>(filter);
174       textField.setTextFormatter(textFormatter);
175     }
176     //
177     if ("File".equals(fieldType) || ("Directory").equals(fieldType)) {
178       if ("File".equals(fieldType))
179         setFileChooser(new FileChooser());
180       else
181         setDirectoryChooser(new DirectoryChooser());
182       button = new Button("...");
183       button.setOnAction(new EventHandler<ActionEvent>() {
184         @Override
185         public void handle(final ActionEvent e) {
186           File file;
187           if ("File".equals(fieldType))
188             file = getFileChooser().showOpenDialog(stage);
189           else
190             file = getDirectoryChooser().showDialog(stage);
191           if (file != null) {
192             textField.setText(file.getPath());
193           }
194         }
195       });
196     }
197     if (field.getHint() != null) {
198       tooltip = new Tooltip();
199       tooltip.setText(field.getHint());
200       control.setTooltip(tooltip);
201     }
202     label = new Label(i18n + ":");
203   }

205   /**
206    * set whether the control is editable
207    * 
208    * @param b
209    */
210   public void setEditable(boolean b) {
211     if (control instanceof TextField) {
212       textField.setEditable(b);
213     } else {
214       control.setDisable(!b);
215     }
216   }

218   /**
219    * set the value of this control
220    * 
221    * @param value
222    */
223   public void setValue(Object value) {
224     if (control instanceof TextField) {
225       if (value != null) {
226         String valueText = value.toString();
227         if ("Double".equals(field.getType())) {
228           Double d = (Double) value;
229           String format=field.getFormat();
230           if (format==null)
231             format="%.2f";
232           valueText = "" + String.format(format, d).trim();
233         }
234         if (debug)
235           LOGGER.log(Level.INFO, field.getId() + "=" + valueText + "("
236               + value.getClass().getSimpleName() + ")");
237         // valueText+="("+updateCount+")";
238         textField.setText(valueText);
239       }
240     } else if (control instanceof CheckBox) {
241       if (value != null)
242         getCheckBox().setSelected((boolean) value);
243       else
244         getCheckBox().setIndeterminate(true);
245     } else if (control instanceof ChoiceBox) {
246       if (value == null) {
247         choiceBox.getSelectionModel().clearSelection();
248       } else {
249         choiceBox.getSelectionModel().select(value.toString());
250       }
251     }
252   }

265   /**
266    * get the value of a control
267    * 
268    * @return the value
269    */
270   public Object getValue() {
271     if (control instanceof TextField) {
272       if ("Integer".equals(field.getType())) {
273         if (textField.getText().isEmpty())
274           return null;
275         Integer result = Integer.parseInt(textField.getText());
276         return result;
277       } else if ("Long".equals(field.getType())) {
278         if (textField.getText().isEmpty())
279           return null;
280         Long result = Long.parseLong(textField.getText());
281         return result;
282       } else if ("Double".equals(field.getType())) {
283         if (textField.getText().isEmpty())
284           return null;
285         // workaround locale issues
286         Double result = Double.parseDouble(textField.getText().replace(',','.'));
287         return result;
288       } else {
289         return textField.getText();
290       }
291     } else if (control instanceof CheckBox) {
292       if (getCheckBox().isIndeterminate())
293         return null;
294       else
295         return getCheckBox().isSelected();
296     } else if (control instanceof ChoiceBox) {
297       String selection=choiceBox.getSelectionModel().getSelectedItem();
298       return selection;
299     }
300     return null;
301   }
