24  package com.bitplan.javafx;
25  
26  import javafx.scene.image.ImageView;
27  import javafx.scene.layout.BorderPane;
28  import javafx.scene.layout.ColumnConstraints;
29  import javafx.scene.layout.GridPane;
30  import javafx.scene.layout.Region;
31  import javafx.scene.layout.RowConstraints;
32  
33  /**
34   * a GridPane with Column and RowConstraints
35   * @author wf
36   *
37   */
38  public class ConstrainedGridPane extends GridPane {
39    /**
40     * helper function for images
41     * @param icon
42     * @return the BoderPane that wraps the given icon
43     */
44    public BorderPane wrapImageView(ImageView icon) {
45      BorderPane pane = new BorderPane();
46      pane.setCenter(icon);
47      // resize images automatically
48      // https://stackoverflow.com/a/12635224/1497139
49      icon.setPreserveRatio(true);
50      icon.fitHeightProperty().bind(pane.heightProperty());
51      icon.fitWidthProperty().bind(pane.widthProperty());
52      return pane;
53    }
54    
55    /**
56     * fix the columnSizes to the given column Width
57     * @param colWidths
58     */
59    public void fixColumnSizes(int hGap,int... colWidths) {
60      this.setHgap(hGap);
61      // Setting columns size in percent
62      for (int colWidth : colWidths) {
63        ColumnConstraints column = new ColumnConstraints();
64        column.setPercentWidth(colWidth);
65        getColumnConstraints().add(column);
66      }
67    }
68  
69    /**
70     * fix the rowSizes to the given rowHeights
71     * @param rowHeights
72     */
73    public void fixRowSizes(int vGap,int... rowHeights) {
74      this.setVgap(vGap);
75      for (int rowHeight : rowHeights) {
76        RowConstraints rowc = new RowConstraints();
77        rowc.setPercentHeight(rowHeight);
78        getRowConstraints().add(rowc);
79      }
80  
81      // grid.setPrefSize(WINDOW_WIDTH, WINDOW_HEIGHT); // Default width and
82      // height
83      this.setMaxSize(Region.USE_COMPUTED_SIZE, Region.USE_COMPUTED_SIZE);
84    }
85  }
