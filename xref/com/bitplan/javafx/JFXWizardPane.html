105   /**
106    * construct me with the given title
107    * 
108    * @param i18nTitle
109    */
110   public JFXWizardPane(JFXWizard wizard,int step, int steps, String i18nTitle) {
111     this.wizard=wizard;
112     fontAwesome = GlyphFontRegistry.font("FontAwesome");
113     if (helpButtonType == null)
114       helpButtonType = new ButtonType(Translator.translate("help"),
115           ButtonBar.ButtonData.HELP);
116     this.setStep(step);
117     this.setSteps(steps);
118     gridPane=new ConstrainedGridPane();
119     gridPane.fixColumnSizes(0, 100);
120     gridPane.fixRowSizes(0,100);
121     setContent(gridPane);
122     this.setI18nTitle(i18nTitle);
123   }

125   /**
126    * set the Help
127    * 
128    * @param help
129    */
130   public void setHelp(String help) {
131     //this.getButtonTypes().add(helpButtonType);
132     //helpButton = this.findButton(helpButtonType);
133     helpButton=new Button(wizard.getI18n("help"));
134     //wizard.getPrivateDialog().getDialogPane().getButtonTypes().add(helpButtonType);
135     //getPrivateButtonBar().getButtons().add(0, helpButton);
136     gridPane.add(helpButton, 0, 1);
137     Glyph helpIcon = fontAwesome.create(FontAwesome.Glyph.QUESTION_CIRCLE);
138     helpButton.setGraphic(helpIcon);
139     
140     helpButton.setOnAction(new EventHandler<ActionEvent>() {
141       @Override
142       public void handle(final ActionEvent actionEvent) {
143         linker.browse(help);
144         actionEvent.consume();
145       }
146     });
147   }

171   /**
172    * fix the encoding problems with controlfx buttons
173    * https://bitbucket.org/controlsfx/controlsfx/issues/769/encoding-problem-all-german-umlauts-are
174    */
175   protected void fixButtons() {
176     ButtonType buttonTypes[] = { ButtonType.NEXT, ButtonType.PREVIOUS,
177         ButtonType.CANCEL, ButtonType.FINISH };
178     Glyph glyphs[] = { fontAwesome.create(FontAwesome.Glyph.CHEVRON_RIGHT),
179         fontAwesome.create(FontAwesome.Glyph.CHEVRON_LEFT),
180         fontAwesome.create(FontAwesome.Glyph.TIMES),
181         fontAwesome.create(FontAwesome.Glyph.CHECK) };
182     int index = 0;
183     for (ButtonType buttonType : buttonTypes) {
184       Button button = findButton(buttonType);
185       if (button != null) {
186         button.setText(buttonType.getText());
187         button.setGraphic(glyphs[index]);
188       }
189       index++;
190     }
191     if (helpButton!=null)
192       helpButton.setText(wizard.getI18n("help"));
193   }

195   /**
196    * get the Button for the given buttonType
197    * 
198    * @return the button
199    */
200   public Button findButton(ButtonType buttonType) {
201     /* Dialog dialog=wizard.getPrivateDialog();
202     return (Button) dialog.getDialogPane().lookupButton(buttonType);
203     */
204    
205     for (Node node : getChildren()) {
206       if (node instanceof ButtonBar) {
207         ButtonBar buttonBar = (ButtonBar) node;
208         ObservableList<Node> buttons = buttonBar.getButtons();
209         for (Node buttonNode : buttons) {
210           Button button = (Button) buttonNode;
211           @SuppressWarnings("unchecked")
212           ObjectProperty<ButtonData> prop = (ObjectProperty<ButtonData>) button
213               .getProperties().get("javafx.scene.control.ButtonBar.ButtonData");
214           ButtonData buttonData = prop.getValue();
215           if (buttonData.equals(buttonType.getButtonData())) {
216             return button;
217           }
218         }
219       }
220     }
221     // LOGGER.log(Level.WARNING,"find Button failed in step "+step+" of "+steps);
222     return null;
223   }

240   /**
241    * load the controller for the given fxmlFileName
242    * @param fxmlFileName
243    * @return - the fxmlFileName
244    */
245   public <T> T load(String fxmlFileName) {
246     LoadResult<T> loadResult=wizard.getFxml().load(fxmlFileName);
247     this.setContentNode(loadResult.parent);
248     controller=loadResult.getController();
249     return loadResult.getController();
250   }

260   /**
261    * set the Content (since setContent of DialogPane is final this function has a different name)
262    * this is a work around to have room for a help button so a grid is interwoven between the dialogpane
263    * and the content we'd like to show
264    * @param parent
265    */
266   public void setContentNode(Node parent) {
267     if (contentNode!=null)
268       gridPane.getChildren().remove(contentNode);
269     gridPane.add(parent, 0, 0);
270     contentNode=parent;
271     // resize the dialog
272     // https://stackoverflow.com/a/31208445/1497139
273     this.wizard.getPrivateDialog().getDialogPane().getScene().getWindow().sizeToScene();
274   }

276   /**
277    * handle an Exception or a Throwable
278    * 
279    * @param th
280    */
281   public void handleException(Throwable th) {
282     load("exception");
283     setI18nTitle("problem_occured");
284     ExceptionController exceptionController = (ExceptionController) controller;
285     exceptionController.handleException(th);
286   }
