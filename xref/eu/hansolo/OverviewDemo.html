763     private static void calcNoOfNodes(Node node) {
764         if (node instanceof Parent) {
765             if (((Parent) node).getChildrenUnmodifiable().size() != 0) {
766                 ObservableList<Node> tempChildren = ((Parent) node).getChildrenUnmodifiable();
767                 noOfNodes += tempChildren.size();
768                 for (Node n : tempChildren) { calcNoOfNodes(n); }
769             }
770         }
771     }
