47  /**
48   * Helper class for LCD Gauges
49   * 
50   * @author wf
51   *
52   */
53  @SuppressWarnings("restriction")
54  public class LcdGauge {
55  
56    public static LcdFont lcdFont = LcdFont.DIGITAL;
57    public static LcdDesign lcdDesign = LcdDesign.GRAY;
58  
59    /**
60     * create the given LCD Gauge
61     * 
62     * @param i18nTitle
63     * @param i18nUnit
64     * @return the gauge
65     */
66    public static Gauge createGauge(String i18nTitle, String i18nUnit,
67        boolean resetable) {
68      return createGaugeLocalized(I18n.get(i18nTitle), I18n.get(i18nUnit),
69          resetable);
70    }
71  
72    /**
73     * create a gauge with the given title and unit
74     * 
75     * @param i18nTitle
76     * @param i18nUnit
77     * @return the Gauge
78     */
79    public static Gauge createGauge(String i18nTitle, String i18nUnit) {
80      return createGauge(i18nTitle, i18nUnit, false);
81    }
82  
83    /**
84     * create a localized gauge
85     * 
86     * @param title
87     * @param unit
88     * @param resetAble
89     *          - should the gauge be resetable?
90     * @return - the gauge
91     */
92    public static Gauge createGaugeLocalized(String title, String unit,
93        boolean resetAble) {
94      Gauge gauge = null;
95      if (resetAble)
96        gauge = new ResetableGauge(title, unit);
97      else
98        gauge = GaugeBuilder.create().skinType(SkinType.LCD)
99            .oldValueVisible(false).maxMeasuredValueVisible(false)
100           .minMeasuredValueVisible(false).decimals(0).tickLabelDecimals(0)
101           .title(title).unit(unit).lcdDesign(lcdDesign).lcdFont(lcdFont)
102           .build();
103     return gauge;
104   }

106   /**
107    * a resetable Gauge Skin
108    * 
109    * @author wf
110    *
111    */
112   public static class ResetableGaugeSkin extends LcdSkin {
113 
114     private Pane pane;
115     // private StackPane stackpane;
116 
117     /**
118      * construct me for the given gauge
119      * 
120      * @param rgauge
121      */
122     public ResetableGaugeSkin(ResetableGauge rgauge) {
123       super(rgauge);
124       pane = (Pane) this.getChildren().get(0);
125       ImageButton rb = rgauge.getResetButton();
126       pane.getChildren().add(rb);
127       rb.translateXProperty().set(5);
128       rb.translateYProperty().set(0);
129       rb.imageHeightProperty().bind(pane.heightProperty().multiply(0.225));
130       rb.imageWidthProperty().bind(pane.widthProperty().multiply(0.15));
131     }
132 
133   }
134 
135   /**
136    * a Gauge with a ResetButton
137    * 
138    * @author wf
139    *
140    */
141   public static class ResetableGauge extends Gauge {
142 
143     private ImageButton resetButton;
144     private ResetableGaugeSkin skin;
145 
146     public ImageButton getResetButton() {
147       return resetButton;
148     }
149 
150     public void setResetButton(ImageButton resetButton) {
151       this.resetButton = resetButton;
152     }
153 
154     /**
155      * construct me
156      * 
157      * @param title
158      * @param unit
159      */
160     public ResetableGauge(String title, String unit) {
161       super(SkinType.LCD);
162       super.oldValueVisibleProperty().set(false);
163       super.maxMeasuredValueVisibleProperty().set(false);
164       super.minMeasuredValueVisibleProperty().set(false);
165       super.tickLabelDecimalsProperty().set(0);
166 
167       super.decimalsProperty().set(0);
168       super.titleProperty().set(title);
169       super.unitProperty().set(unit);
170       super.lcdDesignProperty().set(LcdGauge.lcdDesign);
171       super.lcdFontProperty().set(LcdGauge.lcdFont);
172       initResetButton();
173       skin = new ResetableGaugeSkin(this);
174       this.setSkin(skin);
175     }
176 
177     /**
178      * initialize my reset button
179      */
180     private void initResetButton() {
181       setResetButton(new ImageButton("resetdown.png", "resetup.png"));
182       getResetButton().setOnAction(new EventHandler<ActionEvent>() {
183         @Override
184         public void handle(ActionEvent e) {
185           DoubleProperty vp = ResetableGauge.this.valueProperty();
186           if (vp.isBound()) {
187             // a bound value can not be set
188 
189           } else {
190             setValue(0);
191           }
192         }
193       });
194     }
195 
196     @Override
197     public String getUserAgentStylesheet() {
198       URL u = getClass().getResource("gauge.css");
199       if (u != null)
200         return u.toExternalForm();
201       else
202         return null;
203     }
204 
205   }

207   /**
208    * create a clock with the given title
209    * 
210    * @param i18nTitle - the title (to run thru I18n  before use)
211    *
212    * @return - the created clock
213    */
214   public static Clock createClock(String i18nTitle) {
215     String title = I18n.get(i18nTitle);
216     Locale locale = Translator.getCurrentLocale();
217     if (locale == null)
218       locale = Locale.getDefault();
219     Clock clock = ClockBuilder.create().skinType(ClockSkinType.LCD)
220         .lcdDesign(lcdDesign).title(title).titleVisible(true)
221         .secondsVisible(true).alarmsEnabled(true).dateVisible(false)
222         .locale(locale).build();
223     return clock;
224   }
